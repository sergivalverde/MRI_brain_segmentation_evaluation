948 bool CAnalyzer::falsePositiveRatioTester(int pi_iLesionReference, int pi_iNbLabelsTest, int pi_iNbLabelsRef, int * *pi_ppiOverlapTab, int *pi_piTPRowSumTab, int *pi_piColumnSumTab, double pi_dfBeta, double pi_dfGamma)
949 {
950  bool bRes = false;
951 
953  // Locals variables declarations
954  bool bExit = false;
955  int k = 0;
956  int &iSumOfTPForCurentRow = pi_piTPRowSumTab[pi_iLesionReference];
957  double dfSumWeight = 0.0;
958  double dfRatioOutsideInside = 0.0; // This is the ratio of the outside part of a tested lesion on the total size od this tested lesion.
959  std::vector<std::pair<int, int> > oSortedCollumVector;
960 
962  // Construction of sorted vector for a TP column
963  for(int l=1; l<pi_iNbLabelsTest; ++l)
964  {
965  int iTmpValOfTP = pi_ppiOverlapTab[pi_iLesionReference][l];
966  oSortedCollumVector.push_back( std::pair<int, int>(l, iTmpValOfTP) );
967  }
968  std::sort(oSortedCollumVector.begin(), oSortedCollumVector.end(), pair_decreasing_comparator());
969 
971  // Test in intersection size decreasing order that the regions overlapping the tested lesion are not too much outside of this lesion
972  while(dfSumWeight<pi_dfGamma && k<pi_iNbLabelsRef && !bExit)
973  {
974  dfRatioOutsideInside = (double)(pi_ppiOverlapTab[0][oSortedCollumVector[k].first])/(double)(pi_piColumnSumTab[oSortedCollumVector[k].first]);
975  bExit = dfRatioOutsideInside>pi_dfBeta;
976  if (!bExit)
977  {
978  dfSumWeight += (double)(oSortedCollumVector[k].second)/(double)(iSumOfTPForCurentRow);
979  }
980  ++k;
981  }
982 
984  // Lesion is detected if a sufficient percentage (gamma) of the regions overlapping the tested lesion is not too much outside of this lesion
985  bRes = !bExit; //Almost equivalent, except floating point imprecision, to bRes=dfSumWeight>=pi_dfGamma;
986 
987  return bRes;
